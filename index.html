// ==UserScript==
// @name         Skribbl.io Cheat by bleh2rich
// @namespace    http://tampermonkey.net/
// @version      1.3 - Custom UI
// @description  Enhances Skribbl.io gameplay with modern features
// @author       bleh2rich
// @match        *://skribbl.io/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    console.log("Skribbl.io Cheat v1.3 by bleh2rich Loaded!");

    // --- FEATURES ---
    function drawText(text) { console.log("Draw:", text); alert('Draw Text: ' + text); }
    function autoGuess() { console.log("Guessing..."); alert('Auto Guesser activated.'); }
    const wordDatabase = ['example', 'words']; console.log("DB:", wordDatabase.length);

    // --- SCRIPT UI ---
    function addCheatMenu() {
        const controls = document.createElement('div');
        // Styles moved to CSS if possible, keeping functional styles here
        Object.assign(controls.style, {
            position: 'fixed', bottom: '10px', right: '10px',
            background: 'rgba(18, 18, 31, 0.85)', backdropFilter: 'blur(5px)',
            padding: '15px', zIndex: '10000', borderRadius: '8px',
            border: '1px solid rgba(0, 245, 212, 0.3)',
            boxShadow: '0 0 15px rgba(0, 245, 212, 0.2)',
            color: '#e0e0e0', fontFamily: 'Poppins, sans-serif'
        });
        controls.innerHTML = `
            
Enhancer Menu

            

                 
            

            

                
            
 `;
        document.body.appendChild(controls);
        const drawBtn = document.getElementById('cheat-draw-btn');
        const guessBtn = document.getElementById('cheat-guess-btn');
        const input = document.getElementById('cheat-text-input');
        if(drawBtn) drawBtn.onclick = () => { if(input && input.value) drawText(input.value); };
        if(guessBtn) guessBtn.onclick = autoGuess;
        [drawBtn, guessBtn].forEach(btn => {
            if(btn) {
                 btn.onmouseover = () => { btn.style.backgroundColor = '#00c4a6'; };
                 btn.onmouseout = () => { btn.style.backgroundColor = '#00f5d4'; };
            }
        });
    }
    if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', () => setTimeout(addCheatMenu, 2000)); }
    else { setTimeout(addCheatMenu, 2000); }

    // --- PASTE YOUR ACTUAL SCRIPT LOGIC HERE ---

})();
